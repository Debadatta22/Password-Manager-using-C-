//  PLATFORM: hidden password input (no echo)
// ═══════════════════════════════════════════════
string getHiddenInput(const string& prompt) {
    cout << prompt;
    cout.flush();
    string password;

#ifdef _WIN32
    char ch;
    while ((ch = _getch()) != '\r') {
        if (ch == '\b') {
            if (!password.empty()) { password.pop_back(); cout << "\b \b"; }
        } else { password += ch; cout << '*'; }
    }
    cout << "\n";
#else
    termios oldt, newt;
    tcgetattr(STDIN_FILENO, &oldt);
    newt = oldt;
    newt.c_lflag &= ~(ECHO | ECHOE | ECHOK | ECHONL);
    tcsetattr(STDIN_FILENO, TCSANOW, &newt);
    getline(cin, password);
    tcsetattr(STDIN_FILENO, TCSANOW, &oldt);
    cout << "\n";
#endif
    return password;
}

// ═══════════════════════════════════════════════
//  CRYPTO ENGINE
//  Multi-layer: SHA256-like key derivation +
//  XOR cipher + Caesar rotation + byte shuffle
// ═══════════════════════════════════════════════
class CryptoEngine {
public:
    // Simple but effective key derivation (PBKDF-lite)
    // Produces a 64-byte key from password + salt via repeated mixing
    static vector<uint8_t> deriveKey(const string& password, const string& salt, int iterations = 10000) {
        vector<uint8_t> key(64, 0);
        string combined = password + salt;
        // Seed key
        for (size_t i = 0; i < combined.size(); i++)
            key[i % 64] ^= (uint8_t)combined[i];
        // Stretch via repeated hashing-like mixing
        for (int iter = 0; iter < iterations; iter++) {
            for (int i = 0; i < 64; i++) {
                key[i] = (uint8_t)(((key[i] * 31 + key[(i + 1) % 64]) ^ (uint8_t)(iter * 7 + i)) & 0xFF);
                key[i] ^= key[(i + 7) % 64];
            }
        }
        return key;
    }

    // Hash password for master verification (one-way)
    static string hashPassword(const string& password, const string& salt) {
        vector<uint8_t> key = deriveKey(password, salt, 50000);
        stringstream ss;
        for (auto b : key) ss << hex << setw(2) << setfill('0') << (int)b;
        return ss.str();
    }

    // Generate random salt
    static string generateSalt(int len = 32) {
        static const char chars[] = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!@#$%";
        random_device rd;
        mt19937 gen(rd());
        uniform_int_distribution<> dist(0, sizeof(chars) - 2);
        string salt;
        for (int i = 0; i < len; i++) salt += chars[dist(gen)];
        return salt;
    }

    // Encrypt plaintext with derived key (multi-layer XOR + rotation)
    static string encrypt(const string& plaintext, const vector<uint8_t>& key) {
        string cipher = plaintext;
        int klen = (int)key.size();
        // Layer 1: XOR with key
        for (size_t i = 0; i < cipher.size(); i++)
            cipher[i] ^= key[i % klen];
        // Layer 2: Caesar rotation using key byte
        for (size_t i = 0; i < cipher.size(); i++)
            cipher[i] = (char)((uint8_t)(cipher[i] + key[(i * 3) % klen]) & 0xFF);
        // Layer 3: byte reversal per 8-byte blocks
        for (size_t i = 0; i + 8 <= cipher.size(); i += 8)
            reverse(cipher.begin() + i, cipher.begin() + i + 8);
        // Base64-like encoding to make it printable
        return toBase64(cipher);
    }

    static string decrypt(const string& encoded, const vector<uint8_t>& key) {
        string cipher = fromBase64(encoded);
        int klen = (int)key.size();
        // Reverse Layer 3
        for (size_t i = 0; i + 8 <= cipher.size(); i += 8)
            reverse(cipher.begin() + i, cipher.begin() + i + 8);
        // Reverse Layer 2
        for (size_t i = 0; i < cipher.size(); i++)
            cipher[i] = (char)((uint8_t)(cipher[i] - key[(i * 3) % klen]) & 0xFF);
        // Reverse Layer 1
        for (size_t i = 0; i < cipher.size(); i++)
            cipher[i] ^= key[i % klen];
        return cipher;
    }

private:
    static const string B64_CHARS;

    static string toBase64(const string& in) {
        string out;
        int val = 0, valb = -6;
        for (uint8_t c : in) {
            val = (val << 8) + c;
            valb += 8;
            while (valb >= 0) {
                out.push_back(B64_CHARS[(val >> valb) & 0x3F]);
                valb -= 6;
            }
        }
        if (valb > -6) out.push_back(B64_CHARS[((val << 8) >> (valb + 8)) & 0x3F]);
        while (out.size() % 4) out.push_back('=');
        return out;
    }

    static string fromBase64(const string& in) {
        string out;
        vector<int> T(256, -1);
        for (int i = 0; i < 64; i++) T[(uint8_t)B64_CHARS[i]] = i;
        int val = 0, valb = -8;
        for (uint8_t c : in) {
            if (T[c] == -1) break;
            val = (val << 6) + T[c];
            valb += 6;
            if (valb >= 0) {
                out.push_back((char)((val >> valb) & 0xFF));
                valb -= 8;
            }
        }
        return out;
    }
};
const string CryptoEngine::B64_CHARS =
    "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";

// ═══════════════════════════════════════════════
//  PASSWORD ENTRY (Data Structure)
// ═══════════════════════════════════════════════
class PasswordEntry {
public:
    int    id;
    string title;       // e.g., "Gmail"
    string username;
    string encPassword; // always stored encrypted
    string url;
    string category;    // Work / Personal / Banking / Social / Other
    string notes;
    string createdAt;
    string updatedAt;
    bool   isFavorite;

    PasswordEntry() : id(0), isFavorite(false) {}

    PasswordEntry(int id, string title, string username, string encPwd,
                  string url, string category, string notes,
                  string createdAt, string updatedAt, bool fav)
        : id(id), title(title), username(username), encPassword(encPwd),
          url(url), category(category), notes(notes),
          createdAt(createdAt), updatedAt(updatedAt), isFavorite(fav) {}

    // CSV serialization (delimiter = §)
    string toCSV() const {
        return to_string(id) + "§" + title + "§" + username + "§" + encPassword
             + "§" + url + "§" + category + "§" + notes
             + "§" + createdAt + "§" + updatedAt + "§" + (isFavorite ? "1" : "0");
    }

    static PasswordEntry fromCSV(const string& line) {
        stringstream ss(line);
        string tok;
        vector<string> f;
        while (getline(ss, tok, '\xa7')) f.push_back(tok); // § = \xa7 in UTF-8 sequence; use pipe fallback
        // Fallback: try '|' delimiter
        if (f.size() < 10) {
            f.clear();
            stringstream ss2(line);
            while (getline(ss2, tok, '|')) f.push_back(tok);
        }
        if (f.size() < 10) throw runtime_error("Corrupt entry");
        return PasswordEntry(stoi(f[0]), f[1], f[2], f[3], f[4], f[5], f[6], f[7], f[8], f[9] == "1");
    }

    void displayMasked() const {
        string star = "  │  Password : ********                      │";
        cout << "  ┌──────────────────────────────────────────────┐\n";
        cout << "  │  ID       : " << left << setw(33) << id        << "│\n";
        cout << "  │  Title    : " << left << setw(33) << title     << "│\n";
        cout << "  │  Username : " << left << setw(33) << username  << "│\n";
        cout << star << "\n";
        cout << "  │  URL      : " << left << setw(33) << url       << "│\n";
        cout << "  │  Category : " << left << setw(33) << category  << "│\n";
        cout << "  │  Favorite : " << left << setw(33) << (isFavorite ? "⭐ Yes" : "No") << "│\n";
        cout << "  │  Updated  : " << left << setw(33) << updatedAt << "│\n";
        cout << "  └──────────────────────────────────────────────┘\n";
    }

    void displayRevealed(const string& plainPwd) const {
        cout << "  ┌──────────────────────────────────────────────┐\n";
        cout << "  │  ID       : " << left << setw(33) << id        << "│\n";

// ═══════════════════════════════════════════════
//  BST INDEX (by title, for fast search)
// ═══════════════════════════════════════════════
struct BSTNode {
    string titleKey; // lowercase
    int    entryId;
    BSTNode* left;
    BSTNode* right;
    BSTNode(string k, int id) : titleKey(k), entryId(id), left(nullptr), right(nullptr) {}
};

class TitleIndex {
    BSTNode* root = nullptr;

    BSTNode* insert(BSTNode* n, const string& k, int id) {
        if (!n) return new BSTNode(k, id);
        if (k < n->titleKey) n->left  = insert(n->left,  k, id);
        else                 n->right = insert(n->right, k, id);
        return n;
    }

    void search(BSTNode* n, const string& kw, vector<int>& ids) const {
        if (!n) return;
        if (n->titleKey.find(kw) != string::npos) ids.push_back(n->entryId);
        search(n->left,  kw, ids);
        search(n->right, kw, ids);
    }

    void destroy(BSTNode* n) {
        if (!n) return;
        destroy(n->left); destroy(n->right); delete n;
    }

public:
    ~TitleIndex() { destroy(root); }
    void clear() { destroy(root); root = nullptr; }

    void insert(const string& title, int id) {
        string k = title;
        transform(k.begin(), k.end(), k.begin(), ::tolower);
        root = insert(root, k, id);
    }

    vector<int> search(const string& keyword) const {
        string k = keyword;
        transform(k.begin(), k.end(), k.begin(), ::tolower);
        vector<int> ids;
        search(root, k, ids);
        return ids;
    }
};

// ═══════════════════════════════════════════════
//  PASSWORD STRENGTH CHECKER
// ═══════════════════════════════════════════════
struct StrengthResult { int score; string label; string color; };

StrengthResult checkStrength(const string& pwd) {
    int score = 0;
    if (pwd.size() >= 8)  score++;
    if (pwd.size() >= 12) score++;
    if (pwd.size() >= 16) score++;
    bool hasUpper = false, hasLower = false, hasDigit = false, hasSpecial = false;
    for (char c : pwd) {
        if (isupper(c)) hasUpper = true;
        if (islower(c)) hasLower = true;
        if (isdigit(c)) hasDigit = true;
        if (!isalnum(c)) hasSpecial = true;
    }
    if (hasUpper)   score++;
    if (hasLower)   score++;
    if (hasDigit)   score++;
    if (hasSpecial) score += 2;

    if (score <= 2) return {score, "WEAK",   "31"};   // red
    if (score <= 4) return {score, "FAIR",   "33"};   // yellow
    if (score <= 6) return {score, "GOOD",   "36"};   // cyan
    return              {score, "STRONG", "32"};       // green
}

// ═══════════════════════════════════════════════
//  PASSWORD GENERATOR
// ═══════════════════════════════════════════════
string generatePassword(int length = 16, bool useUpper = true,
                        bool useDigits = true, bool useSpecial = true) {
    string chars = "abcdefghijklmnopqrstuvwxyz";
    if (useUpper)   chars += "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
    if (useDigits)  chars += "0123456789";
    if (useSpecial) chars += "!@#$%^&*()-_=+[]{}|;:,.<>?";

    random_device rd;
    mt19937 gen(rd());
    uniform_int_distribution<> dist(0, (int)chars.size() - 1);

    string pwd;
    for (int i = 0; i < length; i++) pwd += chars[dist(gen)];
    return pwd;
}

// ═══════════════════════════════════════════════
//  SECURE STORAGE (File Handler)
// ═══════════════════════════════════════════════
class SecureStorage {
    string dataFile;
    string metaFile; // stores salt + master hash

public:
    SecureStorage(const string& base) : dataFile(base + ".vault"), metaFile(base + ".meta") {}

    bool metaExists() {
        ifstream f(metaFile);
        return f.good();
    }

    void saveMeta(const string& salt, const string& masterHash) {
        ofstream f(metaFile, ios::trunc);
        if (!f) throw runtime_error("Cannot write meta file.");
        f << salt << "\n" << masterHash << "\n";
    }

    pair<string,string> loadMeta() {
        ifstream f(metaFile);
        if (!f) throw runtime_error("Meta file not found.");
        string salt, hash;
        getline(f, salt);
        getline(f, hash);
        return {salt, hash};
    }

    void saveEntries(const vector<PasswordEntry>& entries) {
        ofstream f(dataFile, ios::trunc);
        if (!f) throw runtime_error("Cannot write vault.");
        for (auto& e : entries) {
            // Use pipe as delimiter (safe since passwords are base64-encoded)
            f << e.id << "|" << e.title << "|" << e.username << "|"
              << e.encPassword << "|" << e.url << "|" << e.category << "|"
              << e.notes << "|" << e.createdAt << "|" << e.updatedAt << "|"
              << (e.isFavorite ? "1" : "0") << "\n";
        }
    }

    vector<PasswordEntry> loadEntries() {
        vector<PasswordEntry> entries;
        ifstream f(dataFile);
        if (!f) return entries;
        string line;
        while (getline(f, line)) {
            if (line.empty()) continue;
            try {
                stringstream ss(line);
                string tok;
                vector<string> flds;
                while (getline(ss, tok, '|')) flds.push_back(tok);
                if (flds.size() < 10) continue;
                PasswordEntry e(stoi(flds[0]), flds[1], flds[2], flds[3],
                                flds[4], flds[5], flds[6], flds[7], flds[8],
                                flds[9] == "1");
                entries.push_back(e);
            } catch (...) {}
        }
        return entries;
    }
};

// ═══════════════════════════════════════════════
//  PASSWORD MANAGER (Core)
// ═══════════════════════════════════════════════
class PasswordManager {
    vector<PasswordEntry> entries;
    TitleIndex bstIndex;
    SecureStorage storage;
    vector<uint8_t> sessionKey;   // derived at login, cleared on lock
    string masterSalt;
    int nextId = 1;
    int failedAttempts = 0;
    static const int MAX_ATTEMPTS = 3;

    string currentTime() {
        time_t now = time(nullptr);
        char buf[20];
        strftime(buf, sizeof(buf), "%Y-%m-%d %H:%M", localtime(&now));
        return string(buf);
    }

    void rebuildIndex() {
        bstIndex.clear();
        for (auto& e : entries) bstIndex.insert(e.title, e.id);
    }

    int findById(int id) {
        for (int i = 0; i < (int)entries.size(); i++)
            if (entries[i].id == id) return i;
        return -1;
    }

public:
    PasswordManager(const string& vaultPath) : storage(vaultPath) {}

    // ─── SETUP MASTER PASSWORD ───
    bool isFirstRun() { return !storage.metaExists(); }

    void setupMaster(const string& masterPwd) {
        masterSalt = CryptoEngine::generateSalt(32);
        string hash = CryptoEngine::hashPassword(masterPwd, masterSalt);
        storage.saveMeta(masterSalt, hash);
        sessionKey = CryptoEngine::deriveKey(masterPwd, masterSalt);
        cout << "\n  ✓ Master password set. Vault initialized.\n";
    }

    // ─── LOGIN ───
    bool login(const string& masterPwd) {
        if (failedAttempts >= MAX_ATTEMPTS) {
            cout << "\n  🔒 LOCKED: Too many failed attempts. Restart app.\n";
            return false;
        }
        auto [salt, hash] = storage.loadMeta();
        masterSalt = salt;
        string inputHash = CryptoEngine::hashPassword(masterPwd, salt);
        if (inputHash != hash) {
            failedAttempts++;
            cout << "  ✗ Wrong password! Attempts left: " << (MAX_ATTEMPTS - failedAttempts) << "\n";
            return false;
        }
        failedAttempts = 0;
        sessionKey = CryptoEngine::deriveKey(masterPwd, salt);
        entries = storage.loadEntries();
        for (auto& e : entries) if (e.id >= nextId) nextId = e.id + 1;
        rebuildIndex();
        return true;
    }

    void lock() {
        sessionKey.clear();
        entries.clear();
        bstIndex.clear();
        nextId = 1;
    }

    bool isLoggedIn() { return !sessionKey.empty(); }

    // ─── CHANGE MASTER PASSWORD ───
    bool changeMaster(const string& oldPwd, const string& newPwd) {
        auto [salt, hash] = storage.loadMeta();
        string inputHash = CryptoEngine::hashPassword(oldPwd, salt);
        if (inputHash != hash) return false;

        // Re-encrypt all entries with new key
        vector<uint8_t> oldKey = sessionKey;
        string newSalt = CryptoEngine::generateSalt(32);
        string newHash = CryptoEngine::hashPassword(newPwd, newSalt);
        vector<uint8_t> newKey = CryptoEngine::deriveKey(newPwd, newSalt);

        for (auto& e : entries) {
            string plain = CryptoEngine::decrypt(e.encPassword, oldKey);
            e.encPassword = CryptoEngine::encrypt(plain, newKey);
        }
        sessionKey = newKey;
        masterSalt = newSalt;
        storage.saveMeta(newSalt, newHash);
        storage.saveEntries(entries);
        return true;
    }

    // ─── ADD ENTRY ───
    void addEntry(const string& title, const string& username,
                  const string& plainPwd, const string& url,
                  const string& category, const string& notes, bool fav) {
        string enc = CryptoEngine::encrypt(plainPwd, sessionKey);
        string ts = currentTime();
        PasswordEntry e(nextId++, title, username, enc, url, category, notes, ts, ts, fav);
        entries.push_back(e);
        bstIndex.insert(title, e.id);
        storage.saveEntries(entries);
    }

    // ─── DELETE ENTRY ───
    bool deleteEntry(int id) {
        int idx = findById(id);
        if (idx < 0) return false;
        entries.erase(entries.begin() + idx);
        rebuildIndex();
        storage.saveEntries(entries);
        return true;
    }

    // ─── UPDATE ENTRY ───
    bool updateEntry(int id, const string& title, const string& username,
                     const string& plainPwd, const string& url,
                     const string& category, const string& notes, bool fav) {
        int idx = findById(id);
        if (idx < 0) return false;
        auto& e = entries[idx];
        e.title    = title;
        e.username = username;
        if (!plainPwd.empty())
            e.encPassword = CryptoEngine::encrypt(plainPwd, sessionKey);
        e.url      = url;
        e.category = category;
        e.notes    = notes;
        e.isFavorite = fav;
        e.updatedAt = currentTime();
        rebuildIndex();
        storage.saveEntries(entries);
        return true;
    }

    // ─── GET DECRYPTED PASSWORD ───
    string revealPassword(int id) {
        int idx = findById(id);
        if (idx < 0) throw runtime_error("Entry not found.");
        return CryptoEngine::decrypt(entries[idx].encPassword, sessionKey);
    }

    // ─── GET ENTRY ───
    PasswordEntry* getById(int id) {
        int idx = findById(id);
        return (idx >= 0) ? &entries[idx] : nullptr;
    }

    // ─── SEARCH BY TITLE (BST) ───
    vector<PasswordEntry> searchByTitle(const string& kw) {
        vector<int> ids = bstIndex.search(kw);
        vector<PasswordEntry> result;
        for (int id : ids) {
            PasswordEntry* e = getById(id);
            if (e) result.push_back(*e);
        }
        return result;
    }

    // ─── SEARCH BY CATEGORY ───
    vector<PasswordEntry> searchByCategory(const string& cat) {
        vector<PasswordEntry> result;
        string cl = cat;
        transform(cl.begin(), cl.end(), cl.begin(), ::tolower);
        for (auto& e : entries) {
            string ec = e.category;
            transform(ec.begin(), ec.end(), ec.begin(), ::tolower);
            if (ec == cl) result.push_back(e);
        }
        return result;
    }

    // ─── FAVORITES ───
    vector<PasswordEntry> getFavorites() {
        vector<PasswordEntry> result;
        for (auto& e : entries) if (e.isFavorite) result.push_back(e);
        return result;
    }

    // ─── ALL ENTRIES (sorted by title) ───
    vector<PasswordEntry> getAllSorted() {
        vector<PasswordEntry> sorted = entries;
        sort(sorted.begin(), sorted.end(), [](const PasswordEntry& a, const PasswordEntry& b) {
            string an = a.title, bn = b.title;
            transform(an.begin(), an.end(), an.begin(), ::tolower);
            transform(bn.begin(), bn.end(), bn.begin(), ::tolower);
            return an < bn;
        });
        return sorted;
    }

    // ─── STATS ───
    map<string,int> categoryStats() {
        map<string,int> stats;
        for (auto& e : entries) stats[e.category]++;
        return stats;
    }

    int totalEntries() { return (int)entries.size(); }

    // ─── WEAK PASSWORD AUDIT ───
    vector<pair<PasswordEntry, StrengthResult>> auditPasswords() {
        vector<pair<PasswordEntry, StrengthResult>> result;
        for (auto& e : entries) {
            string plain = CryptoEngine::decrypt(e.encPassword, sessionKey);
            auto sr = checkStrength(plain);
            if (sr.score <= 4) result.push_back({e, sr});
        }
        return result;
    }

    // ─── EXPORT (encrypted backup) ───
    void exportVault(const string& fname) {
        ofstream f(fname);
        if (!f) throw runtime_error("Cannot create export file.");
        f << "# PASSWORD MANAGER EXPORT — ENCRYPTED\n";
        f << "# Generated: " << currentTime() << "\n";
        f << "# Entries: " << entries.size() << "\n\n";
        for (auto& e : entries) {
            f << "ID=" << e.id << "\n";
            f << "Title=" << e.title << "\n";
            f << "User=" << e.username << "\n";
            f << "Pass=" << e.encPassword << "  [ENCRYPTED]\n";
            f << "URL=" << e.url << "\n";
            f << "Category=" << e.category << "\n";

string inputLine(const string& prompt) {
    cout << "  " << prompt;
    string s;
    getline(cin, s);
    // trim
    size_t st = s.find_first_not_of(" \t\r\n");
    size_t en = s.find_last_not_of(" \t\r\n");
    return (st == string::npos) ? "" : s.substr(st, en - st + 1);
}
